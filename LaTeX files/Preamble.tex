\newcommand{\mline}[1][0pt]{%
  \hspace{-\arraycolsep}%
  \ifdim#1>0pt
    \dimen0=\ht\strutbox \dimen2=\dimen0%chktex 41 %chktex 1
    \advance\dimen0 #1\relax%chktex 41
    \ht\strutbox=\dimen0%chktex 41
  \fi
  \smash{\strut\vrule} % the `\vrule` is as high and deep as a strut
  % since assignments to \ht\strutbox are global, we restore the height
  \ifdim#1>0pt
    \ht\strutbox=\dimen2%chktex 41
  \fi
  \hspace{-\arraycolsep}%
}
